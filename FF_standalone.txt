## FinalFits standalone setup instructions
# NB this also requires having access to CVMFS, though in principle this could probably be dropped too

# Step 1: Install combine and ff dependencies
micromamba env create -n ff-test python=3.10
micromamba activate ff-test
CONDA_SUBDIR=linux-aarch64 micromamba install -c conda-forge pythia8  # You may or may not need to specify the subdir here
micromamba install cms-combine
micromamba install uproot scipy mkl

# Step 2: Setup FinalFits
mkdir src
cd srk
FINALFIT_TAG=higgsdnafinalfit
git clone -b $FINALFIT_TAG https://github.com/cms-analysis/flashggFinalFit.git

# Step 3: Download inputs from the tutorial (can also use `scp` or similar here)
kinit -kt ~/.krb5_auth/.keytab_cern USERNAME
xrdcp --recursive root://eoscms.cern.ch//store/group/phys_higgs/cmshgg/tutorials/HiggsDNA_FinalFits_2024/FinalFits_tutorial/inputs/ .

# Step 4: Lie to children
export CMSSW_BASE=/parent/of/src/flashggFinalFit
mkdir -p $CMSSW_BASE/src/flashggFinalFit/  # Can probably be more clever with this path but for now this works
export PYTHONPATH=$PYTHONPATH:/path/to/flashggFinalFit/tools
export SCRAM_ARCH=None


# Step 5: Download data from the lhc-hxswg (needed for the tutorial's physics model)
# This can be avoided in future by cloning the combine repo and installing directly from there
mkdir -p $CMSSW_BASE/src/HiggsAnalysis/CombinedLimit/data/lhc-hxswg
cd $CMSSW_BASE/src/HiggsAnalysis/CombinedLimit/data/lhc-hxswg
curl -L -o lhc-hxswg.tar.gz "https://github.com/cms-analysis/HiggsAnalysis-CombinedLimit/archive/refs/heads/main.tar.gz"
tar -xzf lhc-hxswg.tar.gz --strip-components=3 HiggsAnalysis-CombinedLimit-main/data/lhc-hxswg
rm lhc-hxswg.tar.gz
cd -
